    genre = models.CharField(max_length=50)
    score = models.IntegerField(default=0)  # increases with interactions

    class Meta:
        unique_together = ('user', 'genre')  # one row per genre per user

Then update preference score on each interaction:
pythondef update_preference(user, book, interaction):
    SCORES = {
        'view': 1,
        'search': 2,
        'like': 5,      # liking weighs more
    }
    pref, _ = UserPreference.objects.get_or_create(user=user, genre=book.genre)
    pref.score += SCORES[interaction]
    pref.save()
